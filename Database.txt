create table boardaccount(
no int(4) not null auto_increment,
id varchar(200) not null,
pw text not null,
nickname varchar(200) not null,
email varchar(250) not null,
registerdate datetime not null default '0001-01-01 01:00:00',
primary key(no, id),
unique key(id,email)
);

create table boardaccess(
no int(8) not null auto_increment,
accessid varchar(200) not null,
accesstime datetime not null default '0001-01-01 01:00:00',
primary key(no)
);
